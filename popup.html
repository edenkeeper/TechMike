Display hostname, ipaddress, wifi, cpu, ram, storage

#10/08/2025

#Script creates full screen windows displaying hostname, ip address and serial and cpu/ram/hdd with a timestamp.
        
        
Add-Type -AssemblyName System.Windows.Forms
Add-Type -AssemblyName System.Drawing

# Gather system info
$hostname = $env:COMPUTERNAME
$ip = (Get-NetIPAddress -AddressFamily IPv4 | Where-Object {$_.InterfaceAlias -notlike "*Loopback*"} | Select-Object -First 1).IPAddress
$cpu = (Get-CimInstance Win32_Processor).Name
$serial = (Get-CimInstance Win32_BIOS).SerialNumber
$ram = "{0:N2} GB" -f ((Get-CimInstance Win32_ComputerSystem).TotalPhysicalMemory / 1GB)

# Fixed storage calculation
$drive = Get-PSDrive | Where-Object { $_.Name -eq 'C' } | Select-Object -First 1
$freeGB = [math]::Round($drive.Free / 1GB, 2)
$totalGB = [math]::Round(($drive.Used + $drive.Free) / 1GB, 2)
$storage = "$freeGB GB Free / $totalGB GB Total"

# Get Wi-Fi SSID and connection type
$wifiInfo = netsh wlan show interfaces
$ssidMatch = $wifiInfo | Select-String '^\s*SSID\s*:\s*(.+)$'
$ssid = if ($ssidMatch) { $ssidMatch.Matches[0].Groups[1].Value.Trim() } else { "Not Connected" }
$connectionType = if ($ssidMatch) { "Wi-Fi" } else { "Ethernet / Wired" }

# Timestamp
$timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"

# Create audit text
$auditText = @"
Hostname: $hostname
IP: $ip
Connection: $connectionType
Wi-Fi SSID: $ssid
CPU: $cpu
RAM: $ram
Storage: $storage
Serial: $serial
Time: $timestamp
"@

# Create form
$form = New-Object System.Windows.Forms.Form
$form.WindowState = 'Maximized'
$form.BackColor = 'Black'
$form.FormBorderStyle = 'None'
$form.TopMost = $true

# Create label
$label = New-Object System.Windows.Forms.Label
$label.Text = $auditText
$label.ForeColor = 'White'
$label.Font = New-Object System.Drawing.Font("Arial", 50, [System.Drawing.FontStyle]::Bold)
$label.Size = New-Object System.Drawing.Size(1600, 900)
$label.Location = New-Object System.Drawing.Point(100, 100)
$label.AutoSize = $false
$label.TextAlign = 'TopLeft'

$form.Controls.Add($label)

# Add keypress event to close on Esc
$form.KeyPreview = $true
$form.Add_KeyDown({
    if ($_.KeyCode -eq 'Escape') {
        $form.Close()
    }
})

$form.ShowDialog()
